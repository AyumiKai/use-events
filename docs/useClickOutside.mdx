# useClickOutside

```js
import { useClickOutside } from 'use-events';
```

```jsx
const Example = () => {
  const [isActive, bind] = useClickOutside(event => console.log(event));

  return (
    <div {...bind}>
      You are {isActive ? 'clicking' : 'not clicking'} outside of this div
    </div>
  );
};
```
