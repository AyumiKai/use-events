# useClickOutside

```js
import { useClickOutside } from 'use-events';
```

```jsx
const Example = () => {
  const ref = React.useRef(null);
  const [isActive] = useClickOutside(ref, event => console.log(event));

  return (
    <div ref={ref}>
      You are {isActive ? 'clicking' : 'not clicking'} outside of this div
    </div>
  );
};
```
