# useActive

```js
import { useActive } from 'use-events';
```

```jsx
const Example = () => {
  const [isActive, bind] = useActive();

  return (
    <div {...bind}>
      You are {isActive ? 'clicking' : 'not clicking'} this div
    </div>
  );
};
```
