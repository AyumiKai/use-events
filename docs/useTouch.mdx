# useTouch

```js
import { useTouch } from 'use-events';
```

```jsx
const Example = () => {
  const [isTouched, bind] = useTouch();

  return (
    <div {...bind}>
      You are {isTouched ? 'touching' : 'not touching'} this div
    </div>
  );
};
```
