# useHover

```js
import { useHover } from 'use-events';
```

```jsx
const Example = () => {
  const [isHovered, bind] = useHover();

  return (
    <div {...bind}>
      You are {isHovered ? 'hovering' : 'not hovering'} this div
    </div>
  );
};
```
